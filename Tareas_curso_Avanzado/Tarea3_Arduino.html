Tarea 3. Expandiendo Arduino

Mi proyecto maneja la memoria interna EEPROM de mi placa de Arduino, haciendo escrituras y lecturas, de datos procedentes de una entrada analógica.

Ademas los comandos para grabar información y leer la información se le dan utilizando un mando IR (de un televisor).

Programo las funcionalidades para los botones:

  1. Apagar --> Muestra datos guardados.
  2. Tecla 0 --> Reset, Inicializa la memoria EEPROM todos los bytes a
  3. Tecla ENTER --> Guarda información del sensor en la EEPROM.

Los demás teclas numéricas también están inicializadas, pero no se les ha asignado ninguna función en concreto.

El montaje es muy simple, tan solo conectar el receptor IR, respetando la polaridad en la alimentación, como sensor, uso un potenciómetro.

#include <IRremote.h>
#include<EEPROM.h>

int RECV_PIN = 11;
IRrecv irrecv(RECV_PIN);
decode_results results;
String DatosEscribir;
int pinIn=0;
int i;
int valor;

void setup()
{
  Serial.begin(9600);
  irrecv.enableIRIn();
}

void loop() {
  if (irrecv.decode(&results)) {
    //Serial.println(results.value, HEX);
    //Serial.println(results.value);
    switch (results.value ) {  
    case 4105841032:
      Serial.println("Mostrar Datos EEPROM \n");

      i=-1;
      valor=10;
      while ((i<512) && (valor>0)){
        i++;
        valor=EEPROM.read(i);
        Serial.print("\t");
        Serial.print(valor,DEC);  
      }

      break;

    case 3778927144:   break;
    case 2908251746:   break;
    case 657459652:    break;
    case 4120482440:   break; 
    case 1931099650:   break;
    case 742730860:    break;
    case 1167253836:   break;
    case 1747313982:   break;
    case 2340753640:   break;

  case 3119867746:
      Serial.println(" \n Reset \n");
      for (i = 0; i < 512; i++) EEPROM.write(i, 0);
      break;

   case 2331063592:
      Serial.println("\n ENTER \n");
      int valor=analogRead(pinIn)/4;
      i=-1; int aux=1;
      while ((i<512) && (aux>0)){
        i++;
        aux=EEPROM.read(i);
      }

      EEPROM.write(i, valor);
      Serial.println("Guardado datos analogicos \n");             
      break;

    }

    irrecv.resume(); 

   }
}
